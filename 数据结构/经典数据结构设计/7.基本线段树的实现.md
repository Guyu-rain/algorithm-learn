# 基本线段树的实现

线段树也是为了解决区间求值的问题，同时支持元素动态更新，解决了前缀和数组的痛点。

代价是时间复杂度稍微高一点，区间查询操作和单点动态更新操作的时间复杂度都是 $O(log n)$。

线段树是一种可以在 $O(log N)$ 的时间复杂度内查询任意长度的区间元素聚合值，在 $O(log ⁡N)$ 的时间复杂度对任意长度的区间元素进行动态修改的**二叉树**，其中 $N$ 为数组中的元素个数。

基本的线段树包含区间查询`query`和单点修改`update`方法。这颗二叉树的所有叶子节点就是数组中的元素，非叶子节点就是索引区间（线段）的汇总信息。

## 链式实现线段树

所谓的链式实现，也就是通过记录左右孩子来索引整棵树。具体如下：

```C++
#include <functional>
#include <iostream>
#include <stdexcept>
#include <vector>

// 线段树节点
class SegmentNode {
public:
    // 该节点表示的区间范围：[l, r]
    int l, r;
    // [l, r] 区间元素的聚合值（比如区间和、区间最大值等）
    int mergeVal;
    SegmentNode* left;
    SegmentNode* right;
    SegmentNode(int mergeVal, int l, int r) : 
        mergetVal(mergeVal), l(l), r(r), left(nullptr), right(nullptr) {}
};

class SegmentTree {
private:
    SegmentNode* root;
    // 定义某种聚合函数
    function<int(int, int)> merger;

    // 定义：将 nums[l..r] 中的元素构建成线段树，返回根节点
    SegmentNode* build(const vector<int>& nums, int l, int r) {
        // 区间只有一个元素，直接返回
        if (l == r) {
            return new SegmentNode(nums[l], l, r);
        }
        // 从中间切分，递归构建左右子树
        int mid = l + (r - l) / 2;
        SegmentNode* left = build(nums, l, mid);
        SegmentNode* right = build(nums, mid + 1, r);
        // 根据左右节点的聚合值，计算跟节点的聚合值
        SegmentNode* node = new SegmentNode(merger(left->mergeVal, right->mergeVal), l, r);
        // 组装左右子树
        node->left = left;
        node->right = right;
        return node;
    }

    void update(SegmentNode* node, int index, int value) {
        // 找到了目标节点，更新值
        if (node->l == node->r) {
            node->mergeVal = value;
            return;
        }

        int mid = node->l + (node->r - node->l) / 2;
        if (index <= mid) {
            // 如果 index 较小，去左子树更新
            update(node->left, index, value);
        } else {
            // 如果 index 较大，去右子树更新
            update(node->right, index, value);
        }

        // 后序位置，左右子树已经更新完毕，更新当前节点的聚合值
        node->mergeVal = merger(node->left->mergeVal, node->right->mergeVal);
    }

    int query(SegmentNode* node, int qL, int qR) {
        if (qL > qR) {
            throw invalid_argument("Invalid query range";)
        }

        // 命中了目标区间，注意：这里是真正的base case，是必须要有的
        if (node->l == qL && node->r == qR) {
            return node->mergeVal;
        }

        // 未命中目标区间，向下寻找
        int mid = node->l + (node->r - node->l) / 2;
        if (qR <= mid) {
            // node.l <= qL <= qR <= mid
            // 目标区间完全在左子树中
            return query(node->left, qL, qR);
        } else if (qL > mid) {
            // mid < qL <= qR <= node.r
            // 目标区间完全在右子树中
            return query(node->right, qL, qR);
        } else {
            // node->l <= qL <= mid < qR <= node->r
            // 目标区间横跨左右子树，将查询区间拆分成两部分，分别向左右子树查询
            // 最后将左右子树的查询结果合并
            return merger(
                query(node->left, qL, mid), 
                query(node->right, mid + 1, qR)
            );
        }
    }
public:
    // 创建线段树
    // 输入数组 nums 和一个聚合函数 merger，后者用于计算区间函数的聚合值
    SegmentTree(const vector<int>& nums, function<int(int, int)> merger) 
    : merger(merger) {
        root = build(nums, 0, nums.size() - 1);
    }

    void update(int index, int value) {
        update(root, index, value);
    }

    int query(int qL, int qR) {
        return query(root, qL, qR);
    }
};
```

## 数组实现线段树

树可以通过数组来实现，链式结构由于需要额外存储左右孩子的指针，因此效率略低。

注意：在初始化线段树时，如果数组中有$n$个元素，线段树叶子节点必然是$n$个。但是线段树不一定是满二叉树，我们一般用 $4\times n$ 这样一个上界来存储线段树。总之，遇到线段树，要多开一部分空间，一般就开 `4 * n`。

下面来实现线段树。我们认为跟节点下标为`0`，其中有两个工具函数，分别是索引左右孩子的。

```C++
class ArraySegmentTree {
    vector<int> tree;
    // 元素个数
    int n;
    function<int(int, int)> merger;

    // 通过 vector nums 构建一颗线段树
    void build(const vector<int>& nums, int l, int r, int rootIndex) {
        // base case，当左右区间收缩到只有一个元素时
        // 此时已经到达了叶子节点，要理解这一点！
        if (l == r) {
            // 区间中只有一个元素，将其设置为叶子节点
            tree[rootIndex] = nums[l];
            return;
        }

        // 从中间切分，左右递归构造线段树
        int mid = l + (r - l) / 2;
        int leftRootIndex = leftChild(rootIndex);
        int rightRootIndex = rightChild(rootIndex);
        build(nums, l, mid, leftRootIndex);
        build(nums, mid + 1, r, rightRootIndex);

        // 后续位置，左右子树已经构建完毕，处理当前节点聚合值 merge 的结果
        tree[rootIndex] = merger(tree[leftRootIndex], tree[rightRootIndex]);
    }

    // 更新二叉树的叶子节点 index 为 val
    void update(int l, int r, int rootIndex, int index, int value) {
        // 当前节点为 rootIndex，对应区间是 [l, r]
        // base case
        // 找到了对应的根节点，将其修改 
        if (l == r) {
            tree[rootIndex] = value;
            return;
        }

        // 递归线段树，去找 index 的位置
        int mid = l + (r - l) / 2;
        if (index <= mid) {
            // 去左子树上找
            update(l, mid, leftChild(rootIndex), index, value);
        } else {
            // 去右子树上找
            update(mid + 1, r, rightChild(rootIndex), index, value);
        }
        // 此时，左右子树都已更新完毕，更新当前节点
        tree[rootIndex] = merger(tree[leftChild(rootIndex)], tree[rightChild(rootIndex)]);
    }

    int query(int l, int r, int rootIndex, int qL, int qR) {
        // base case 
        // 如果命中区间，直接返回
        if (qL == l && qR == r) {
            return tree[rootIndex];
        }

        // 没有命中，递归搜索整棵树
        int mid = l + (r - l) / 2;
        int leftRootChild = leftChild(rootIndex);
        int rightRootChild = rightChild(rootIndex);
        if (qR <= mid) {
            // 搜索区间完全在左子树上
            return query(l, mid, leftRootChild, qL, qR);
        } else if (qL >= mid) {
            // 搜索区间完全在右子树上
            return query(mid + 1, r, rightRootChild, qL, qR);
        } else {
            // 搜索区间横跨左右子树
            // 注意！这里需要修改左右查找区间
            return merger(
                query(qL, mid, leftRootChild, qL, mid),
                query(mid + 1, qR, rightRootChild, mid + 1, qR)
            );
        }
    }
// 两个辅助函数
    int leftChild(int pos) {
        return 2 * pos + 1;
    }

    int rightChild(int pos) {
        return 2 * pos + 2;
    }

public:
    ArraySegmentTree(const vector<int>& nums, function<int(int, int)> merger) {
        this->n = nums.size();
        this->merger = merger;
        // 将树构造起来
        // 注意；一定要先开一块空间
        tree.resize(4 * n);
        build(nums, 0, n - 1, 0);
    }

    void update(int index, int value) {
        update(0, n - 1, 0, index, value);
    }

    int query(int qL, int qR) {
        // 判断是否合法
        if (qL < 0 || qR >= n || qL < qR) {
            throw invalid_argument("Invalid range: [" + to_string(qL) + ", " + to_string(qR) + "]");
        }
        return query(0, n - 1, 0, qL, qR);
    }
};
```
