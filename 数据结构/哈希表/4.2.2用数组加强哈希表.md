# 用数组加强哈希表（ArrayHashMap）

我们用数组维护哈希表的所有`keys`，这样做的好处是`keys`变得稠密，可以用这个性质进行随机挑选等操作。

加入一个`key`很容易，但如何在 `O(1)` 时间内删除数组中的任意元素？

通过交换交换要删除的key和最后一个元素，然后永远删除最后一个元素，这样能够避免元素的搬动。

```C++
template<typename K, typename V>
class ArrayHashMap {
    struct Node {
        K key;
        V val;

        Node(K key, V val) : key(key), val(val) {}
    };

    // 存储 key 和 key 在 arr 中的索引
    unordered_map<K, int> map;
    // 真正存储 key-value 的数组
    vector<Node> arr;
    // 随机数生成器
    default_random_engine e;

public:
    ArrayHashMap() {
        e.seed(random_device()());
    }

    V get(K key) {
        if (!map.count(key)) {
            return NULL;
        }
        int index = map[key];
        return arr[index].val;
    }

    void put(K key, V val) {
        if (containsKey(key)) {
            int i = map[key];
            arr[i].val = val;
            return;
        }
        arr.push_back(Node(key, val));
        map[key] = arr.size() - 1;
    }

    void remove(K key) {
        if (!map.count(key)) {
            return;
        }
        int index = map[key];
        Node node = arr[index];
        // 1. 最后一个元素 e 和 index个 元素交换位置
        Node e = arr.back();
        swap(arr[index], arr.back());

        // 2. 修改 map 中 e.key 对应的索引
        map[e.key] = index;

        // 3. 在数组中删除最后一个元素
        arr.pop_back();

        // 4. 在 map 中删除 node.key
        map.erase(node.key);
    }

    // 随机弹出一个键
    K randomKey() {
        int n = arr.size();
        uniform_int_distribution<int> u(0, n - 1);
        int randomIndex = u(e);
        return arr[randomIndex].key;
    }

    bool containsKey(K key) {
        return map.count(key);
    }

    int size() {
        return map.size();
    }
};
```
